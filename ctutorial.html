3. Variables and Data Types

                        
        #include <stdio.h>
        
        int main() {
            int age = 25;
            float pi = 3.14;
            char letter = 'A';
            
            printf("Age: %d\n", age);
            printf("Pi: %.2f\n", pi);
            printf("Letter: %c\n", letter);
            
            return 0;
        }

Explanation: This program declares variables of different data types (integer, float, and char) and prints their values using the printf function. The output will display the age as an integer, the value of Pi with two decimal places, and a character letter.

4. Operators in C

                        
        #include <stdio.h>

            int a = 10, b = 5;
            printf("Addition: %d\n", a + b);
            printf("Subtraction: %d\n", a - b);
            printf("Multiplication: %d\n", a * b);
            printf("Division: %d\n", a / b);
            printf("Modulus: %d\n", a % b);
        
            return 0;
        }

Explanation: This program performs basic arithmetic operations like addition, subtraction, multiplication, division, and modulus on two integers (a and b), and prints the results using printf.

5. Control Flow Statements

                        
        #include <stdio.h>
        
        int main() {
            int x = 10;
            
            // if-else statement
            if (x > 5) {
                printf("x is greater than 5\n");
            } else {
                printf("x is less than or equal to 5\n");
            }
        
            // switch statement
            switch(x) {
                case 10:
                    printf("x is 10\n");
                    break;
                default:
                    printf("x is not 10\n");
            }
        
            // loop
            for(int i = 0; i < 3; i++) {
                printf("Iteration: %d\n", i);
            }
        
            return 0;
        }

Explanation: This program demonstrates conditional and loop control structures in C. It uses an if-else statement to check if x is greater than 5, a switch statement to check if x equals 10, and a for loop to iterate and print values from 0 to 2.

7. Arrays and Strings

                        
        #include <stdio.h>
        
        int main() {
            // Array of integers
            int arr[3] = {1, 2, 3};
            
            // Print array elements
            for(int i = 0; i < 3; i++) {
                printf("arr[%d] = %d\n", i, arr[i]);
            }
        
            // String handling
            char str[] = "Hello, C!";
            printf("String: %s\n", str);
            
            return 0;
        }

Explanation: This program demonstrates working with arrays and strings in C. It defines an integer array `arr` and prints its elements using a `for` loop. It also defines a string `str` and prints it using `printf`.

10. File Handling in C

                        
        #include <stdio.h>
        
        int main() {
            FILE *file = fopen("test.txt", "w");
        
            if (file == NULL) {
                printf("Could not open file\n");
                return 1;
            }
        
            fprintf(file, "This is a test file.\n");
            fclose(file);
        
            file = fopen("test.txt", "r");
            char line[100];
            fgets(line, sizeof(line), file);
            printf("File contents: %s", line);
            fclose(file);
        
            return 0;
        }

Explanation: This program demonstrates file handling in C. It opens a file called "test.txt" for writing, writes a line of text to it, then closes the file. It reopens the file in read mode, reads the contents, and prints them to the console.

11. Memory Management


        #include <stdio.h>
        #include <stdlib.h>
        
        int main() {
            int *arr = (int *)malloc(5 * sizeof(int));
        
            if (arr == NULL) {
                printf("Memory allocation failed\n");
                return 1;
            }
        
            for (int i = 0; i < 5; i++) {
                arr[i] = i + 1;
            }
        
            for (int i = 0; i < 5; i++) {
                printf("arr[%d] = %d\n", i, arr[i]);
            }
        
            free(arr);
        
            return 0;
        }

Explanation: This program demonstrates dynamic memory allocation in C using `malloc`. It allocates memory for an array of 5 integers, assigns values to the array, prints them, and then frees the allocated memory using `free`. Other memory allocation functions in C include:

  • `calloc`: Allocates memory for an array and initializes it to zero.
    Example: `int *arr = (int *)calloc(5, sizeof(int));`
  • `realloc`: Changes the size of a previously allocated memory block.
    Example: `arr = (int *)realloc(arr, new_size);`
